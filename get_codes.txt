class Payment{
    public:
    virtual void paymentProcess(double amount) = 0;

    void printReceipt(double amount){
        cout<< "Receipt generated."<<endl;
        cout<< "You have paid "<< amount<< endl;
    }
};
class CreditCardPayment : public Payment{
    public:
    void paymentProcess(double amount)override{
        if(amount >  0){
            printReceipt(amount);
        }
    }
};

    public:

        if(amount > 0){
            printReceipt(amount);
        }
    }
};
