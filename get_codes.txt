#include <iostream>
#include <string>
using namespace std;

class Player{
    protected:
    string playerName;
    int jerseyNumber;

    public:

    Player(string playerName, int jerseyNumber){
        this->playerName = playerName;
        this->jerseyNumber = jerseyNumber;
    }

    virtual string displayInfo(){
        string info = playerName + " " + to_string(jerseyNumber);
        return info;
    }
    // pair<string, int> displayInfo(){
    //     return {playerName, jerseyNumber};
    // }
};
class Footballer : public Player {
    private:
    string position;

    public:
    Footballer(string playerName, int jerseyNumber, string position) : Player(playerName, jerseyNumber){
        this->position = position;
    }

    string displayInfo()override{
        string info = playerName + " " + to_string(jerseyNumber) + " " + position;
        return info;
    }

    
};
class Cricketer : public Player{
    private:
    bool isWicketKeeper;

    public:
    Cricketer(string playerName , int jerseyNumber, bool isWicketKeeper) : Player(playerName , jerseyNumber){
        this->isWicketKeeper = isWicketKeeper;
    }

    string displayInfo()override{
        string info = playerName + " " + to_string(jerseyNumber) + " " + to_string(isWicketKeeper);
        return info;
    }
};
